214 " From an idea by Michael Naumann
215 function! VisualSearch(direction) range
216     let l:saved_reg = @"
217     execute "normal! vgvy"
218
219     let l:pattern = escape(@", '\\/.*$^~[]')
220     let l:pattern = substitute(l:pattern, "\n$", "", "")
221
222     if a:direction == 'b'
223         execute "normal ?" . l:pattern . "^M"
224     elseif a:direction == 'gv'
225         call CmdLine("vimgrep " . '/'. l:pattern . '/' . ' **/*.')
226     elseif a:direction == 'f'
227         execute "normal /" . l:pattern . "^M"
228     endif
229
230     let @/ = l:pattern
231     let @" = l:saved_reg
232 endfunction

268 func! DeleteTillSlash()
269   let g:cmd = getcmdline()
270   if MySys() == "linux" || MySys() == "mac"
271     let g:cmd_edited = substitute(g:cmd, "\\(.*\[/\]\\).*", "\\1", "")
272   else
273     let g:cmd_edited = substitute(g:cmd, "\\(.*\[\\\\]\\).*", "\\1", "")
274   endif
275   if g:cmd == g:cmd_edited
276     if MySys() == "linux" || MySys() == "mac"
277       let g:cmd_edited = substitute(g:cmd, "\\(.*\[/\]\\).*/", "\\1", "")
278     else
279       let g:cmd_edited = substitute(g:cmd, "\\(.*\[\\\\\]\\).*\[\\\\\]", "\\1", "")
280     endif
281   endif  
282   return g:cmd_edited
283 endfunc
